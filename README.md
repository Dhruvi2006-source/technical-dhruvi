export const useEventNotifier = (eventId) => {
  const [eventData, setEventData] = useState(null);

  useEffect(() => {
    socket.emit("join-event-room", eventId);

    socket.on("event-update", (data) => {
      setEventData(data);
    });

    return () => socket.emit("leave-event-room", eventId);
  }, [eventId]);

  return eventData;
};
